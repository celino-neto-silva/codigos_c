Faça um algoritmo para receber um número qualquer e:
-imprimir na tela se o número é par ou ímpar
-positivo ou negativo.

inicio algoritmo
    inteiro var num
    leia num

    caractere var resp1,resp2

    se(num < 0)
        resp1 = "Negativo"
    senao
        resp1 = "Positivo"
    fimse

    se(num % 2 = 0)
        resp2 = "Par"
    senao
        resp2 = "Ímpar"
    fimse

    imprima "O numero(num) é: (resp1) e (resp2)"
fim algoritmo
